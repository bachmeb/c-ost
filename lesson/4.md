##### Make a directory
    mkdir -p ~/git/c-ost/four

##### Change to the directory
    cd ~/git/c-ost/four

##### Make a c file
    vim four.c
```
/* four.c  */
#include <stdio.h>

int main(){
  int days;

  days = 1;

  printf("I've been programming C for %i day(s)\n",days);

  return(1);
}
```
##### Compile
    gcc -o four four.c

##### Run the application
    ./four
